\newglossaryentry{pickandplace}{
    name={pick-and-place},
    description={A robotic operation in which an object is picked up from one position and placed at another}
}

\newglossaryentry{scanpose}{
    name={scan pose},
    description={The predefined arm configuration used when the camera observes the workspace for object detection}
}

\newglossaryentry{workspace}{
    name={workspace},
    description={The physical area that the robotic arm can reach and operate within}
}

\newglossaryentry{gripper}{
    name={gripper},
    description={The end-effector mechanism used to grasp and release objects}
}

\newglossaryentry{homography}{
    name={homography},
    description={A perspective transformation used to map image coordinates to workspace coordinates}
}

\newglossaryentry{ensembledetection}{
    name={ensemble detection},
    description={A detection approach that combines outputs from multiple detection methods into a unified result}
}

\newglossaryentry{tracking}{
    name={object tracking},
    description={The process of maintaining stable identities for detected objects across consecutive frames}
}

\newglossaryentry{kalmanfilter}{
    name={Kalman filter},
    description={A recursive estimation algorithm used to predict and smooth object motion over time}
}

\newglossaryentry{touchcalibration}{
    name={touch calibration},
    description={A calibration procedure where the arm physically touches reference points to map workspace coordinates}
}

\newglossaryentry{sagcompensation}{
    name={sag compensation},
    description={A correction method used to compensate for gravitational droop in the robotic arm}
}

\newglossaryentry{closedform}{
    name={closed-form solution},
    description={A mathematical solution computed directly through equations without iterative approximation}
}

\newglossaryentry{realtimesystem}{
    name={real-time system},
    description={A system whose correctness depends on both logical output and timing constraints}
}

\newglossaryentry{coasting}{
    name={coasting},
    description={A tracking mode in which an object continues using predicted motion when detections are temporarily unavailable}
}

\newglossaryentry{forwardkinematics}{
    name={forward kinematics},
    description={The computation of Cartesian position from known joint angles}
}

\newglossaryentry{inversekinematics}{
    name={inverse kinematics},
    description={The computation of joint angles required to reach a desired Cartesian position}
}

\newglossaryentry{daisychain}{
    name={daisy-chain},
    description={A wiring topology in which devices are connected sequentially along the same communication and power line}
}

\newglossaryentry{falsepositive}{
    name={false positive},
    description={A detection incorrectly identified as a valid object}
}

\newglossaryentry{falsenegative}{
    name={false negative},
    description={A valid object that the detection system fails to detect}
}
